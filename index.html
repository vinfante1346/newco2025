Code Example: BaaS Abstraction Layer

// baas-router.service.ts
@Injectable()
export class BaasRouterService {
  constructor(
    private alviereService: AlviereService,
    private galileoService: GalileoService,
    private migrationTrackerService: MigrationTrackerService
  ) {}

  async getAccount(userId: string): Promise {
    const migrationStatus = await this.migrationTrackerService
      .getStatus(userId);

    if (migrationStatus === 'MIGRATED_TO_GALILEO') {
      return this.galileoService.getAccount(userId);
    } else {
      return this.alviereService.getAccount(userId);
    }
  }

  async createTransaction(userId: string, txData: TransactionDto) {
    const migrationStatus = await this.migrationTrackerService
      .getStatus(userId);

    if (migrationStatus === 'MIGRATED_TO_GALILEO') {
      return this.galileoService.createTransaction(txData);
    } else {
      return this.alviereService.createTransaction(txData);
    }
  }
}
